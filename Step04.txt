function textSplit() {    var title = document.querySelector('.title');    var splitText = title.textContent.split('');    title.textContent = '';    var character;    for(var i=0, len=splitText.length; i<len; i++) {        character = document.createElement('span');        character.textContent = splitText[i];        character.style.display = 'inline-block';        chars.push(character);        title.appendChild(character);    }}
